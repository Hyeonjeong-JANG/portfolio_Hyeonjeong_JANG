                "value"

          }
        }, this);
      },

    Number.prototype.formatNumber = function () {
      if (this == 0) return 0;
      let regex = /(^[+-]?\d+)(\d{3})/;
      let nstr = this + "";

      return nstr;
    };

프로젝트 기획 배경

과거에 비해 집에 있는 시간이 늘어나면서 집을 나만의 '힐링 공간'으로 꾸미고자 하는 문화가 자리잡았다.
'조명'은 공간을 크게 차지하지 않으면서도 집안 전체의 분위기를 확 바뀌게 하는 인테리어 소품으로 각광받고 있다.
최근 전구 생선 전문 회사였던 일광전구가 '조명 기구 생산'으로 사업 영역 확대를 하였지만 홈페이지의 디자인이 경쟁 업체에 비해 트렌디하지 못하고 사용성이 떨어져서 중계 업체를 통해서 구입해야만 하는 번거로움이 있다.

프로젝트 상세 소개

리뉴얼 콘셉트 및 기대 효과

  • 12column grid를 활용한 레이아웃 - 시각적 편안함 제공
  • 직관성이 높은 카테고리 분류 - 구매 가이드라인 제시
  • 편의 기능의 적재적소 배치 - 사용성 향상

타겟 유저

  • 일터이자 안식처를 나만의 취향으로 채워가는 20~30대 재택 근무자 (+ 재택 근무를 겸하는 직장인)
  • 창업을 시작하며 가게 공간을 꾸미게 된 신규 창업자 및 소상공인
